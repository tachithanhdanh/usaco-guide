## C++ Implementation

```cpp
#include <bits/stdc++.h>

using namespace std;

int main(){
	freopen("div7.in", "r", stdin);
	freopen("div7.out", "w", stdout);
	int n;
	cin >> n;
	vector<int> pref(n+1);
	int ans=0;
	vector<int> first(7, -1);
	first[0]=0;
	for(int i=1;i<=n;i++){
		int x;
		cin >> x;
		pref[i]=(pref[i-1]+x)%7;
		if(first[pref[i]]==-1){
			first[pref[i]]=i;
		}else{
			ans=max(ans, i-first[pref[i]]);
		}
	}
	cout << ans << endl;
}
```
